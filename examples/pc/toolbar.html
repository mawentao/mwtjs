工具栏

工具栏用于放置一些操作控件。

类名：mwt.ToolBar

配置属性

示例：单行工具栏


var option = [ {value:0,text:"男"},{value:1,text:"女"}];
var citys = [
			{text:"北京",value:1},
			{text:"新疆齐齐哈尔市",value:101},
			{text:"上海",value:2},
			{text:"广州",value:3},
			{text:"深圳",value:4},
			{text:"南京",value:5},
			{text:"杭州",value:6},
			{text:"武汉",value:7},
			{text:"成都",value:8},
			{text:"青岛",value:9},
			{text:"重庆",value:10},
			{text:"大连",value:11},
			{text:"厦门",value:12},
			{text:"兰州",value:13},
			{text:"常州",value:14},
			{text:"无锡",value:15},
			{text:"徐州",value:16},
			{text:"镇江",value:17},
			{text:"苏州",value:18}
];
new mwt.ToolBar({
    render : 'toolbar-div-01',
    items  : [
        '',
        {label:"add",cls:'mwt-btn-success',handler:function(){alert("add");}},
        {type:'select',id:"sex1",label:"性别",width:50,value:'1',options:option,handler:query},
		{type:'selbox',id:"city",label:"城市",width:'100px',popWidth:100,value:'1',options:citys,handler:query},
        {type:'radiobtn',id:'sex2',value:'1',width:50,cls:'mwt-btn-danger',options:option,handler:query},
        {type:'search',id:'sokey',width:200,placeholder:'查询关键词',handler:query},
        {type:"datepicker",id:'dt',label:"选择日期",value:-1,handler:query},
        '->',
        {type:"daterangepicker",id:'dt2',label:"选择日期区间",value:-30,handler:query}
    ]
}).create();

function query()
{
    var params = {
        sex1: mwt.get_select_value('sex1'),
        sex2: mwt.get_value('sex2'),
		city: mwt.get_value('city'),
        sokey: mwt.get_value('sokey'),
        day: mwt.get_value('dt'),
        dayrange: mwt.get_value('dt2')
    }
    print_r(params);
}

示例：多行工具栏


var option = [ {value:0,text:"男"},{value:1,text:"女"}];
new mwt.ToolBar({
    render : 'toolbar-div-02',
    style  : 'padding:10px;border-radius:4px;background:#FDFAF0;border:solid 1px #FFECB3;',
    items  : [
        '',
        {label:"add",cls:'mwt-btn-success',handler:function(){alert("add");}},
        {type:'select',label:"性别",width:50,value:'1',options:option},
        '-',
        {type:"daterangepicker",label:"选择日期区间",value:-7},
        '-',
        {type:'radiobtn',value:'1',width:50,cls:'mwt-btn-danger',options:option},
        {type:'search',width:200,placeholder:'查询关键词'},
        '->',
        {type:"datepicker",label:"选择日期",value:0}
    ]
}).create();
