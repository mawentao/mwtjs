Border布局

Border布局将整个容器分为5个部分：east(东)、south(南)、west(西)、north(北)和center(中)。

不可伸缩不可折叠的布局


new mwt.BorderLayout({
    render : 'border-div-1',
    items : [
        {id:'frame-north', region:'north', height:30,style:'background:#DEF;border-bottom:solid 1px #ccc',html:'north'},
        {id:'frame-south', region:'south', height:30,style:'background:#DEF;border-top:solid 1px #ccc',html:'south'},
        {id:'frame-west',  region:'west',  width:50,style:'background:#FEE;border-right:solid 1px #ccc',html:'west'},
        {id:'frame-east',  region:'east',  width:50,style:'background:#FEE;border-left:solid 1px #ccc',html:'east'},
        {id:'frame-center',region:'center',style:'background:#FDF5DD;',html:'center'}
    ]
}).init();

可折叠的布局


new mwt.BorderLayout({
    render : 'border-div-2',
    items : [
        {region:'north',height:30,collapsible:true,style:'background:#DEF',html:'north'},
        {region:'south',height:30,collapsible:true,style:'background:#DEF',html:'south'},
        {region:'west',width:50,collapsible:true,style:'background:#FEE',html:'west'},
        {region:'east',width:50,collapsible:true,style:'background:#FEE',html:'east'},
        {region:'center',style:'background:#FDF5DD;',html:'center'}
    ]
}).init();

可伸缩可折叠的布局


new mwt.BorderLayout({
    render : 'border-div-3',
    items : [
        {region:'north',height:30,collapsible:true,split:true,style:'background:#DEF',html:'north'},
        {region:'south',height:30,collapsible:true,split:true,style:'background:#DEF',html:'south'},
        {region:'west',width:50,collapsible:true,split:true,style:'background:#FEE',html:'west'},
        {region:'east',width:50,collapsible:true,split:true,style:'background:#FEE',html:'east'},
        {region:'center',style:'background:#FDF5DD;',html:'center'}
    ]
}).init();
